#The below powershell script is used to Force all users to change their password at next logon - You can also force reset password for a specific OU only by entering the correct OU name for example 'SLT'#
Get-ADUser -Filter * -SearchBase “CN=Users,DC=Dormers,DC=local” | Set-ADUser -CannotChangePassword:$false -PasswordNeverExpires:$false -ChangePasswordAtLogon:$true
